mysql常用查询

查看数据库中是否有某个名字的表SHOW TABLES LIKE '%tablename%'

删除数据表

drop table tblname;

插入数据

insert into tbl_name (id, name) values(1, "heke");

where子句不区分大小写where binary区分大小写

update修改更新数据

update tbl_name set name="yunkehe" where id=1;

删除记录

delete from tbl_name where name="frank";

like语句,获取以frank结尾的记录

select * from tbl_name where name like "%frank";

排序, asc/desc

select * from tbl_name oreder by id;

分组

select name, count(*) from employee_tbl group by name;
select name, sum(singin) as single_count from employee_tbl group by name with rollup;
select coalesce(name, "总数"), sum(singin) as single_count from employee_tbl group by name with rollup;

多表联查

join select a.id, a.name, b.count from tbl_a a join tbl_b b on a.name = b.name;
